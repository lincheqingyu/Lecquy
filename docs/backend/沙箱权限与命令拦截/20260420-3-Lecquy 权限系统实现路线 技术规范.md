```typescript
/**
 * 权限模式 (Permission Mode)
 * 控制 AI 命令执行的总体策略
 */
export const PERMISSION_MODES = [
  'default',          // 默认：每次操作都询问用户
  'dontAsk',          // 已知操作不再询问
  'plan',             // 计划模式：仅预览，不执行
  'acceptEdits',      // 自动接受所有编辑
  'bypassPermissions' // 绕过所有权限检查
] as const

export type PermissionMode = typeof PERMISSION_MODES[number]

/**
 * 权限行为 (Permission Behavior)
 * 单条规则的执行行为
 */
export type PermissionBehavior = 'allow' | 'deny' | 'ask'

/**
 * 权限规则来源
 */
export type PermissionRuleSource =
  | 'userSettings'     // ~/.lecquy/settings.json
  | 'projectSettings'  // project/.lecquy/settings.json
  | 'sessionSettings'  // 当前会话
  | 'cliArg'          // 命令行参数

/**
 * 单条权限规则
 */
export type PermissionRule = {
  source: PermissionRuleSource
  behavior: PermissionBehavior
  toolName: string
  content?: string  // 可选：例如文件路径模式、命令模式
}

/**
 * 权限检查结果
 */
export type PermissionResult =
  | { decision: 'allow'; reason: string }
  | { decision: 'deny'; reason: string }
  | { decision: 'ask'; reason: string }
  | { decision: 'plan'; reason: string }
```

#### 1.2 实现权限加载器

**文件**：`backend/src/runtime/permissions/loader.ts`

```typescript
/**
 * 从多个源加载权限规则
 */
export async function loadPermissionRules(options: {
  userDir?: string
  projectDir?: string
  cliArgs?: string[]
}): Promise<PermissionRule[]> {
  const rules: PermissionRule[] = []

  // 1. 加载用户设置 (最低优先级)
  if (options.userDir) {
    const userRules = await loadFromFile(
      join(options.userDir, '.lecquy', 'permissions.json')
    )
    rules.push(...userRules.map(r => ({ ...r, source: 'userSettings' })))
  }

  // 2. 加载项目设置
  if (options.projectDir) {
    const projectRules = await loadFromFile(
      join(options.projectDir, '.lecquy', 'permissions.json')
    )
    rules.push(...projectRules.map(r => ({ ...r, source: 'projectSettings' })))
  }

  // 3. 加载 CLI 参数
  if (options.cliArgs) {
    const cliRules = parseCliArgs(options.cliArgs)
    rules.push(...cliRules)
  }

  // 4. 按优先级排序 (CLI > 项目 > 用户)
  return sortByPriority(rules)
}

/**
 * 优先级排序 (高到低)
 */
function sortByPriority(rules: PermissionRule[]): PermissionRule[] {
  const priority = {
    'cliArg': 3,
    'sessionSettings': 2,
    'projectSettings': 1,
    'userSettings': 0
  }
  
  return rules.sort((a, b) => priority[b.source] - priority[a.source])
}
```

#### 1.3 实现权限检查引擎

**文件**：`backend/src/runtime/permissions/checker.ts`

```typescript
/**
 * 核心权限检查函数
 */
export async function checkPermission(
  toolName: string,
  context: {
    mode: PermissionMode
    rules: PermissionRule[]
    currentPath?: string
    userId?: string
  }
): Promise<PermissionResult> {
  // 1. 查找匹配的规则
  const matchedRule = findMatchingRule(toolName, context.rules)

  // 2. 检查拒绝规则 (最高优先级)
  if (matchedRule?.behavior === 'deny') {
    return {
      decision: 'deny',
      reason: `Tool '${toolName}' is denied by ${matchedRule.source}`
    }
  }

  // 3. 检查允许规则
  if (matchedRule?.behavior === 'allow') {
    return { decision: 'allow', reason: 'Allowed by rule' }
  }

  // 4. 根据权限模式判断
  switch (context.mode) {
    case 'bypassPermissions':
      return { decision: 'allow', reason: 'Bypass mode enabled' }
    
    case 'acceptEdits':
      return { decision: 'allow', reason: 'Auto-accept mode' }
    
    case 'plan':
      return { decision: 'plan', reason: 'Plan mode - preview only' }
    
    case 'dontAsk':
      // 如果没有明确的规则，在 dontAsk 模式下拒绝
      return { decision: 'deny', reason: 'Operation not in approved list' }
    
    case 'default':
    default:
      return { decision: 'ask', reason: 'User confirmation required' }
  }
}

/**
 * 查找匹配规则
 */
function findMatchingRule(
  toolName: string,
  rules: PermissionRule[]
): PermissionRule | undefined {
  // 精确匹配优先
  for (const rule of rules) {
    if (rule.toolName === toolName) {
      return rule
    }
  }
  
  // 通配符匹配
  for (const rule of rules) {
    if (rule.toolName === '*') {
      return rule
    }
  }
  
  return undefined
}
```

```typescript
export const DANGEROUS_DIRECTORIES = [
  '.git',
  '.vscode',
  '.idea',
  '.lecquy'

/**
 * 检查是否为危险文件
 */
export function isDangerousFile(filePath: string): boolean {
  const basename = getBasename(filePath)
  return DANGEROUS_FILES.includes(basename as any)
}

/**
 * 检查是否在危险目录中
 */
export function isInDangerousDirectory(filePath: string): boolean {
  for (const dangerDir of DANGEROUS_DIRECTORIES) {
    if (filePath.includes(`/${dangerDir}/`) || filePath.includes(`\\${dangerDir}\\`)) {
      return true
    }
  }
  return false
}
```

#### 2.2 文件操作权限检查

**文件**：`backend/src/runtime/permissions/file-operations.ts`

```typescript
/**
 * 检查文件编辑权限
 */
export async function checkFileEditPermission(
  filePath: string,
  context: {
    rules: PermissionRule[]
    mode: PermissionMode
  }
): Promise<PermissionResult> {
  // 1. 检查危险文件
  if (isDangerousFile(filePath)) {
    return {
      decision: 'ask',  // 危险文件需要用户确认
      reason: `${filePath} is a protected system file`
    }
  }

  // 2. 检查危险目录
  if (isInDangerousDirectory(filePath)) {
    return {
      decision: 'ask',
      reason: `${filePath} is in a protected directory`
    }
  }

  // 3. 检查路径遍历攻击
  if (containsPathTraversal(filePath)) {
    return {
      decision: 'deny',
      reason: 'Path traversal detected'
    }
  }

  // 4. 常规权限检查
  return checkPermission('FileEditTool', context)
}

/**
 * 检查文件删除权限
 */
export async function checkFileDeletePermission(
  filePath: string,
  context: {
    rules: PermissionRule[]
    mode: PermissionMode
  }
): Promise<PermissionResult> {
  // 删除比编辑更严格 - 危险文件直接拒绝
  if (isDangerousFile(filePath) || isInDangerousDirectory(filePath)) {
    return {
      decision: 'deny',
      reason: `Cannot delete protected file: ${filePath}`
    }
  }

  return checkFileEditPermission(filePath, context)
}

/**
 * 路径遍历检测
 */
function containsPathTraversal(path: string): boolean {
  const normalized = normalize(path)
  return normalized.includes('..') || path.includes('../')
}

function normalize(path: string): string {
  return path.replace(/\\/g, '/')
}
```

#### 3.1 Bash 命令危险等级分类

**文件**：`backend/src/runtime/permissions/bash-classifier.ts`

```typescript
/**
 * Bash 命令危险等级分类
 * MVP 版本使用规则匹配，后续可升级为 AI 分类
 */

export type CommandRiskLevel = 'safe' | 'ask' | 'deny'

/**
 * 规则定义的危险命令
 */
const DENY_PATTERNS = [
  // 文件系统销毁
  /^rm\s+(-[a-z]*r|-[a-z]*f)*\s+\//,
  /^dd\s+/,
  /^mkfs/,
  
  // 进程炸弹
  /fork\(\)\s*&&\s*fork\(\)/,
  /:\(\)\s*{\s*:\|:\s*&\s*};\s*:/,
  
  // 网络扫描
  /^nmap\s+/,
  /^nc\s+-l/,
  
  // 权限提升滥用
  /^sudo\s+/,
]

const ASK_PATTERNS = [
  // 网络操作
  /^(curl|wget|nc|telnet|ssh)/,
  
  // 包管理
  /^(apt|yum|brew|pacman|pip|npm)\s+(install|remove|update)/,
  
  // 系统修改
  /^(systemctl|service)\s+/,
  /^chown\s+/,
  /^chmod\s+/,
]

/**
 * 分类 Bash 命令
 */
export function classifyBashCommand(command: string): CommandRiskLevel {
  // 1. 检查拒绝模式（绝对禁止）
  for (const pattern of DENY_PATTERNS) {
    if (pattern.test(command)) {
      return 'deny'
    }
  }

  // 2. 检查询问模式（需要确认）
  for (const pattern of ASK_PATTERNS) {
    if (pattern.test(command)) {
      return 'ask'
    }
  }

  // 3. 默认为安全
  return 'safe'
}

/**
 * 获取命令的风险说明
 */
export function getCommandRiskReason(
  command: string,
  riskLevel: CommandRiskLevel
): string {
  switch (riskLevel) {
    case 'deny':
      return `Command may cause system damage: "${command}"`
    case 'ask':
      return `Command requires explicit confirmation: "${command}"`
    case 'safe':
      return `Command is safe to execute: "${command}"`
  }
}
```

#### 3.2 Bash 工具权限集成

**文件**：`backend/src/runtime/permissions/bash-tool.ts`

```typescript
/**
 * 检查 Bash 命令权限
 */
export async function checkBashCommandPermission(
  command: string,
  context: {
    rules: PermissionRule[]
    mode: PermissionMode
    cwd: string
  }
): Promise<PermissionResult> {
  // 1. 分类命令风险等级
  const riskLevel = classifyBashCommand(command)
  
  // 2. 基于风险等级决策
  switch (riskLevel) {
    case 'deny':
      return {
        decision: 'deny',
        reason: getCommandRiskReason(command, 'deny')
      }
    
    case 'ask':
      // 检查是否有专门的规则
      const rule = findMatchingRule('bash', context.rules)
      if (rule?.behavior === 'allow') {
        return { decision: 'allow', reason: 'Allowed by rule' }
      }
      // 否则询问用户
      return {
        decision: 'ask',
        reason: getCommandRiskReason(command, 'ask')
      }
    
    case 'safe':
      // 常规权限检查
      return checkPermission('bash', context)
  }
}
```

**文件**：`backend/src/runtime/permissions/updater.ts`

```typescript
/**
 * 权限规则更新操作
 */
export type PermissionUpdate =
  | { type: 'addRule'; rule: PermissionRule }
  | { type: 'removeRule'; toolName: string; source: PermissionRuleSource }
  | { type: 'clearRules'; source: PermissionRuleSource }

/**
 * 应用权限更新
 */
export async function applyPermissionUpdate(
  update: PermissionUpdate,
  targetDir: string
): Promise<void> {
  const configPath = join(targetDir, '.lecquy', 'permissions.json')
  
  // 1. 加载当前规则
  const config = await loadJsonFile(configPath)
  const rules = config.rules || []

  // 2. 应用更新
  let updatedRules: PermissionRule[]
  
  switch (update.type) {
    case 'addRule':
      updatedRules = [...rules, update.rule]
      break
    
    case 'removeRule':
      updatedRules = rules.filter(r =>
        !(r.toolName === update.toolName && r.source === update.source)
      )
      break
    
    case 'clearRules':
      updatedRules = rules.filter(r => r.source !== update.source)
      break
  }

  // 3. 保存到文件
  const newConfig = { ...config, rules: updatedRules }
  await saveJsonFile(configPath, newConfig, { pretty: true })
}
```

## 6. 关键决策文档

### 6.1 为什么简化为 5 种权限模式？

- ✅ **覆盖 80% 使用场景**
- ✅ **更容易解释给用户**
- ✅ **减少状态管理复杂性**
- ✅ 后续可扩展到 7 种

简化清单：
- 移除 `bubble` 模式（内部实现细节）
- 合并 `auto` 到 `default` 的条件分支（可升级）

### 6.2 为什么 MVP 先用规则分类，后用 AI？

**时间成本考量**：
- 规则匹配：实现 1-2 天，覆盖 70% 危险命令
- AI 分类：需要调用 API，延迟和成本，但准确率更高

**分阶段策略**：
```
第 1 阶段（现在）：规则 + 明确拒绝
         ↓
第 2 阶段（后续）：规则 + 侧查询分类器
         ↓
第 3 阶段（成熟）：优先级 AI，规则作为后备
```

```typescript
type PermissionMode = 'default' | 'dontAsk' | 'plan' | 'acceptEdits'
type PermissionBehavior = 'allow' | 'deny' | 'ask'

interface Rule { tool: string; behavior: PermissionBehavior }

function checkPermission(
  tool: string,
  mode: PermissionMode,
  rules: Rule[]
): 'allow' | 'deny' | 'ask' {
  const rule = rules.find(r => r.tool === tool)
  if (rule?.behavior === 'deny') return 'deny'
  if (mode === 'acceptEdits') return 'allow'
  if (mode === 'plan') return 'ask'  // 预览模式
  return rule?.behavior ?? 'ask'
}
```
